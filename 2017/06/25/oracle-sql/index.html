Oracle常用SQL

2017-06-25
database

–查看索引
SELECT STATUS, T.* FROM USER_INDEXES T

–INDEX SIZE

SELECT SEGMENT_NAME,
  (SUM(BYTES) / 1024 / 1024) || 'MB' AS INDEX_SIZE
  FROM USER_SEGMENTS
  WHERE SEGMENT_TYPE = 'INDEX'
  GROUP BY SEGMENT_NAME;

–TABLE SIZE

SELECT SEGMENT_NAME, (SUM(BYTES) / 1024 / 1024) || 'MB' AS TABLE_SIZE
  FROM DBA_EXTENTS
  WHERE SEGMENT_TYPE = 'TABLE'
  GROUP BY SEGMENT_NAME
  HAVING SUM(BYTES) / (1024 * 1024) > 1

–索引碎片分析，会锁定整张表
ANALYZE INDEX xx_ORDERID_INDEX VALIDATE STRUCTURE

–索引碎片比率 (DEL_LF_ROWS_LEN/LF_ROWS_LEN)*100，超过20%需要整理碎片
SELECT HEIGHT, DEL_LF_ROWS / LF_ROWS FROM INDEX_STATS

–所有表空间倒排

SELECT S.BYTES, S.*
  FROM DBA_SEGMENTS S
  WHERE S.BYTES IS NOT NULL
  ORDER BY S.BYTES DESC

–所有表行数倒排

SELECT T.NUM_ROWS, T.*
  FROM DBA_TABLES T
  WHERE T.NUM_ROWS IS NOT NULL
  ORDER BY T.NUM_ROWS DESC

 SELECT T.NUM_ROWS, T.*
    FROM DBA_TABLES T
    ORDER BY T.NUM_ROWS DESC

–重新统计, 执行计划的准确性（主要指数据返回，数据量大小）由统计信息的准确性决定
ANALYZE TABLE xx COMPUTE STATISTICS;

–查看视图字段类型

SELECT *
    FROM USER_TAB_COLS A
    WHERE A.TABLE_NAME LIKE 'R%'

–表的字段名称

SELECT COLUMN_NAME,
  TABLE_NAME,
  DATA_TYPE,
  DATA_LENGTH,
  DATA_PRECISION,
  DATA_SCALE
  FROM USER_TAB_COLUMNS
  WHERE TABLE_NAME = 'xx'

–表空间

SELECT 'xx' DBNAME,
  B.FILE_ID 文件ID,
  B.TABLESPACE_NAME 表空间,
  B.FILE_NAME 物理文件名,
  ROUND(B.BYTES / 1024 / 1024 / 1024, 2) 总数G,
  ROUND((B.BYTES - SUM(NVL(A.BYTES, 0))) / 1024 / 1024 / 1024,2) 已使用G,
  ROUND(SUM(NVL(A.BYTES, 0)) / 1024 / 1024 / 1024, 2) 剩余G,
  ROUND(SUM(NVL(A.BYTES, 0)) / (B.BYTES), 2) 剩余百分比
  FROM DBA_FREE_SPACE A, DBA_DATA_FILES B
  WHERE A.FILE_ID = B.FILE_ID
  GROUP BY B.TABLESPACE_NAME,
  B.FILE_NAME,
  B.FILE_ID,
  B.BYTES
  HAVING ROUND(SUM(NVL(A.BYTES, 0)) / (B.BYTES), 2) < 0.1

–临时表资源使用SQL

SELECT SE.USERNAME,
  SE.SID,
  SU.EXTENTS,
  SU.BLOCKS * TO_NUMBER(RTRIM(P.VALUE)) ASSPACE,
  TABLESPACE,
  SEGTYPE,
  SQL_TEXT
  FROM V$SORT_USAGE SU,
  V$PARAMETER P,
  V$SESSION SE,
  V$SQL S
 WHERE P.NAME = 'DB_BLOCK_SIZE'
  AND SU.SESSION_ADDR = SE.SADDR
  AND S.HASH_VALUE = SU.SQLHASH
  AND S.ADDRESS = SU.SQLADDR
 ORDER BY SE.USERNAME, SE.SID;

–查看低效执行的SQL

SELECT EXECUTIONS,
  DISK_READS,
  BUFFER_GETS,
  ROUND((BUFFER_GETS - DISK_READS) / BUFFER_GETS, 2) HIT_RADIO,
  ROUND(DISK_READS / EXECUTIONS, 2) READS_PER_RUN,
  SQL_TEXT
  FROM V$SQLAREA
 WHERE EXECUTIONS > 0
  AND BUFFER_GETS > 0
  AND (BUFFER_GETS - DISK_READS) / BUFFER_GETS < 0.8
 ORDER BY 4 DESC;

–最后修改时间

SELECT UAT.TABLE_NAME AS 表名,
  (SELECT LAST_DDL_TIME
  FROM USER_OBJECTS
  WHERE OBJECT_NAME = UAT.TABLE_NAME) AS 最后修改日期
  FROM USER_ALL_TABLES UAT

–所有对象创建修改时间
SELECT CREATED, OBJECT_NAME, LAST_DDL_TIME FROM USER_OBJECTS;

–最近执行的SQL

SELECT SQL_TEXT, LAST_LOAD_TIME
  FROM V$SQL
 ORDER BY LAST_LOAD_TIME DESC
